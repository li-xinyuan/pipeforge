```python
@router.post("/orchestrate")
async def orchestrate(req: AiSuggestionRequest, request: Request):
    """AI plans multi-step SQL pipeline from natural language."""
    _check_rate_limit(request.client.host if request.client else "unknown")
    settings = load_settings()
    if not settings.enabled:
        raise HTTPException(status_code=400, detail="AI 未配置，请先在设置中启用")
    backend = None
    try:
        backend = create_backend(settings)
        prompt = build_prompt("orchestrate", req.context)
        logger.info("AI orchestrate request prompt_len=%d model=%s", len(prompt), settings.model)
        start = time.monotonic()
        result = await asyncio.wait_for(backend.generate(prompt), timeout=90.0)
        latency_ms = int((time.monotonic() - start) * 1000)
        logger.info("AI orchestrate response latency_ms=%d response_len=%d", latency_ms, len(result))
        # Try to parse as JSON; if fails, return raw with parse error flag
        try:
            parsed = json.loads(parse_response(result))
            return {"steps": parsed.get("steps", []), "explanation": parsed.get("explanation", ""), "raw": result}
        except (json.JSONDecodeError, KeyError):
            return {"steps": [], "explanation": "", "raw": result, "parse_error": True}
    except asyncio.TimeoutError:
        raise HTTPException(status_code=503, detail="AI 响应超时，请稍后重试")
    except Exception as e:
        msg = str(e)
        logger.error("AI orchestrate failed error=%s", msg[:200])
        raise HTTPException(status_code=500, detail="AI 调用失败，请稍后重试")
    finally:
        if backend is not None:
            await backend.close()
```

```typescript
// configforge-web/src/composables/useWizardApi.ts — add to useAiApi

async function askOrchestrate(context: Record<string, unknown>): Promise<{
  steps: Array<{ name: string; input_tables: string[]; output_tables: string[]; sql: string }>;
  explanation: string;
  raw?: string;
  parse_error?: boolean;
} | null> {
  error.value = null
  try {
    const resp = await fetch('/api/ai/orchestrate', {
      method: 'POST',
      headers: { 'Content-Type': 'application/json' },
      body: JSON.stringify({ category: 'orchestrate', context }),
    })
    if (!resp.ok) {
      const err = await resp.json()
      error.value = err
      return null
    }
    return await resp.json()
  } catch (e) {
    error.value = { error: e instanceof Error ? e.message : '网络错误' }
    return null
  }
}
```

```typescript
async function onOrchestrate(naturalLanguage: string) {
  const context: Record<string, unknown> = {
    inputs: store.inputs.map(inp => {
      const meta = store.uploadedFiles[inp.fileId]
      return { table: inp.table, columns: meta?.columns || [] }
    }),
    outputColumns: (store.output?.config?.columns || []).map(c => c.target),
    naturalLanguage,
  }

  const result = await askOrchestrate(context)
  if (result && result.steps.length > 0) {
    aiMessages.value.push({
      role: 'ai',
      content: '已规划处理链：',
      orchestration: result,
    })
  } else if (result?.parse_error) {
    aiMessages.value.push({
      role: 'ai',
      content: 'AI 返回格式异常，请重试或手动配置。原始响应：' + (result.raw || '').slice(0, 500),
    })
  } else {
    aiMessages.value.push({
      role: 'ai',
      content: '无法规划处理链，请检查输入源是否已上传文件并加载列信息。',
    })
  }
}
```

```python
    result = engine.execute(params, log_dir=LOG_DIR)
except Exception as e:
    # Try to capture which step failed and what tables exist
    error_msg = str(e)
    # Extract column info for SQL errors
    column_hint = ""
    if "no such column" in error_msg.lower():
        import re
        match = re.search(r"no such column: (\S+)", error_msg, re.IGNORECASE)
        if match:
            bad_col = match.group(1)
            # Try to get available columns from context
            try:
                available_cols = context.db.get_column_names(proc_spec.input_tables[0]) if proc_spec.input_tables else []
                if available_cols:
                    column_hint = f" — 可用列：{', '.join(available_cols)}"
            except Exception:
                pass
    raise RuntimeError(json.dumps({
        "error": error_msg + column_hint,
        "failed_step": failed_step,
```

```typescript
function setError(err: { error?: string; code?: string; detail?: string; failed_step?: number; intermediate_tables?: string[] } | string | null) {
  if (!err) { clearAll(); return }
  if (typeof err === 'string') {
    errorMessage.value = err
    return
  }
  errorCode.value = err.code || null
  errorDetail.value = err.detail || null
  failedStep.value = err.failed_step ?? null
  intermediateTables.value = err.intermediate_tables || []
  errorMessage.value = (err.code && ERROR_MAP[err.code]) || err.error || '操作失败，请稍后重试'
}
```
